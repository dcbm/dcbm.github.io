A little Vis of all sights so visit in vienna.

The data comes from publicly available data, found on: https://data.wien.gv.at.

library(magrittr)

read.csv("https://data.wien.gv.at/daten/geo?service=WFS&request=GetFeature&version=1.1.0&typeName=ogdwien:SEHENSWUERDIGOGD&srsName=EPSG:4326&outputFormat=csv") %>% 
  dplyr::mutate(COORDS = stringr::str_remove(SHAPE, "POINT "),
                COORDS = stringr::str_remove(COORDS, "\\("),
                COORDS = stringr::str_remove(COORDS, "\\)"),
                X = (stringr::str_split(COORDS, " ") %>% purrr::transpose())[[1]] %>% unlist() %>% as.numeric(),
                Y = (stringr::str_split(COORDS, " ") %>% purrr::transpose())[[2]] %>% unlist() %>% as.numeric()) %>% 
  dplyr::select(NAME, ADRESSE, WEITERE_INF, THUMBNAIL, X, Y) %>% 
  leaflet::leaflet(width = "100%") %>% 
  leaflet::addProviderTiles(leaflet::providers$BasemapAT.orthofoto) %>% 
  # leaflet::addMarkers(~X,~Y, 
  #                     label =  ~paste0(NAME),
  #                     popup = ~paste0( "<a href='",WEITERE_INF,"'>",NAME,"</a>", "<br/>",ADRESSE),
  #                     icon = ~leaflet::icons(iconUrl = THUMBNAIL))
  leaflet::addMarkers(~X,~Y, 
                      label =  ~NAME,
                      popup = ~paste0( "<img src='", THUMBNAIL,"'alt=''>", "<br/>",
                        "<a href='",WEITERE_INF,"'>",NAME,"</a>", "<br/>", ADRESSE))
